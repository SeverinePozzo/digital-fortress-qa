CREATE TABLE raw_users (
    user_id INT,
    email VARCHAR(255),
    created_at TIMESTAMP
);

CREATE TABLE raw_orders (
    order_id VARCHAR(50),
    user_id INT,
    order_amount DECIMAL(10,2),
    order_date DATE
);

CREATE TABLE raw_payments (
    payment_id INT,
    order_id VARCHAR(50),
    payment_status VARCHAR(50),
    amount_paid DECIMAL(10,2)
);
